##并查集

将所有相关的事物归并到同一个集合里，这个集合称为并查集。
用处：
用Kruskal算法求最小生成树。
主要步骤：

  1. 将每个点赋初值，先开始每个点自成一个集合，其父亲就是自己本身；
  2. 查找元素所在的集合，主要是找其祖先；
  3. 合并两个集合，将一个集合的祖先指向另一个集合；

并查集的优化：
查找时候可以使路径压缩，及当两个元素的祖先是一个的时候，可以将该元素直接指向其祖先，而不必通过层层递进，指向其祖先。
合并的优化，因为将元素少的集合合并到元素多的集合中，这样形成的树的高度会比较小；
使用附加的数组记录子树的大小；

Poj 1251
求最小生成树
用Kruskal算法做
代码如下

#include<iostream>
#include<algorithm>
using namespace std;
const int MAX = 26;
struct Village
{
 int x;//记录点
 int y;//记录点
 int pow;//记录权值
};

Village v[MAX*MAX];
int rank[MAX];
int root[MAX];
bool cmp(Village v1,Village v2)//结构体比较函数
{
 if(v1.pow<v2.pow)
 {
  return true;
 }
 else
  return false;
}
int Find(int x)//查找根
{
 if(root[x]!=x)
 {
  root[x] = Find(root[x]);
 }
 return root[x];
}
int sum = 0;
void Union( int x,int y,int pow )//合并集合
{
 if(x==y)
  return ;//合并的两个集合必定不相等
 if(rank[x]<rank[y])//将个数小的集合并到个数大的集合中去；
 {
          root[x] = y; 
 }
 else
 {
         if(rank[x]==rank[y])
   {
              rank[x]++;//因为上面的x，y是通过find找到的根的，所以已经经过了压缩，可看成只有一个度的元素；只需在这里将rank加
   }
   root[y] = x;
 }
 sum+=pow;
}
int main()
{
 int n,count;
 while(cin>>n&&n)
 {
          int m;
    char c1,c2;
    count = 0;
    for(int i = 0;i<n;i++)
    {
     root[i] = i;//此处就是给所有的点初始化，使所有的点的父亲都是自己本身。
     rank[i] = 0;//因为开始的子树个数都是0；
    }

    for(int i = 0;i<n-1;i++)
    {
     cin>>c1>>m;    
     for(int j = 0;j < m;j++)
     {
      cin>>c2>>v[count].pow;
      v[count].x = c1 - 'A';
      v[count].y = c2 - 'A';
      count++;
     }
    }
  sort(v,v+count,cmp);//将权值小的路径先连
 /* for(int i = 0;i<count;i++)
  {
   char c1,c2;
   c1 = v[i].x+'A';
   c2 = v[i].y+'A';
   cout<<c1<<"->"<<v[i].pow<<"->"<<c2<<endl;
  }*/
  sum = 0;
  for(int i = 0;i<count;i++)
  {
   Union(Find(v[i].x),Find(v[i].y),v[i].pow);
  }
  cout<<sum<<endl;
 }
 return 0;
}
