poj1556 Bellman_Ford

发表于 2012-03-09   |   分类于 ACM   |  

##poj 1556

这一题的题意是说给你一堵墙的一个横坐标，以及四个纵坐标，要求出从起始点（0，5）到终点（10，5）的最短路径。注意，当一堵墙将连线遮住的时候则该路径不可走。
这题难的是构图，我们将墙上的每个点连线，还需要判断是否可以连线，并统计其距离。再用最短路径算法求最短路。

#include

#include

#include

#include

using namespace std;
const int MAXN = 20;
const int MAXP = 100;
const double INF = INT_MAX / 2;

struct Point
{
double x,y;
};
Point p[MAXP];

double wx[MAXN],py[MAXN][4];

int N,M,Pn;

double Dis(Point a, Point b)
{
return sqrt((a.x - b.x)(a.x - b.x) + (a.y - b.y)(a.y - b.y));
}

double Cross(Point a,Point b,double x,double y) // x,y是否在point a，b的连线上或下。
{
return (a.x - x)(a.y - b.y) - (a.y - y)(a.x-b.x);
}

bool IsLine(Point a,Point b)
{
if(a.x >= b.x) return false;
int i = 0;
bool flag = true;
while(a.x>=wx[i]&&i<N)
i++;
while(wx[i]<b.x&&i<N)
{
if(Cross(a,b,wx[i],0)Cross(a,b,wx[i],py[i][0])<0||Cross(a,b,wx[i],py[i][1])Cross(a,b,wx[i],py[i][2])<0||Cross(a,b,wx[i],py[i][3])*Cross(a,b,wx[i],10)<0)
{
flag = false;
break;
}
i++;
}
return flag;
}

struct Node
{
int to,next;
double value;
};
Node n[MAXP*MAXP];

int indiex;
int head[MAXP];

void make_map(int from,int to,double value)
{
n[indiex].to = to;
n[indiex].value = value;
n[indiex].next = head[from];
head[from] = indiex++;
}

void Set_Map()
{
M = 0;
for(int i = 0;i < Pn;i++)
{
for(int j = i+1;j < Pn;j++)
{
if(IsLine(p[i],p[j]))
{
//cout<<i<<’ ‘<<j<<endl;
make_map(i,j,Dis(p[i],p[j]));
M++;
}
}
}
}

double Bellman_Ford()
{
double d[MAXP];
d[0] = 0;
for(int i = 1;i < Pn;i++)
{
d[i] = INF;
}
for(int i = head[0];~i;i=n[i].next)
{
d[n[i].to] = n[i].value;
}

bool ex = true;

for(int k = 1;k < Pn&&ex;k++)
{
    ex = false;
    for(int i = 1;i < Pn;i++)
    {
        for(int j = head[i];~j;j=n[j].next)
        {
            int v = n[j].to;
            if(d[i]<INF&&d[i]+n[j].value < d[v])
            {
                d[v] = d[i] + n[j].value;
                ex = true;
            }
        }
    }
}

return d[Pn-1];

}

int main()
{
//freopen(“in.txt”,”r”,stdin);
while(scanf(“%d”,&N)&&~N)
{
indiex = 0;
memset(head,-1,sizeof(head));
p[0].x = 0;
p[0].y = 5;
p[4N+1].x = 10;
p[4N+1].y = 5;
Pn = 4*N+2;

    for(int i = 0;i < N;i++)
    {
        scanf("%lf%lf%lf%lf%lf",&wx[i],&py[i][0],&py[i][1],&py[i][2],&py[i][3]);
        for(int j = 1;j<=4;j++)
        {
            p[4*i+j].x = wx[i];
            p[4*i+j].y = py[i][j-1];
        }
    }
    Set_Map();
    double Ans = Bellman_Ford();
    printf("%.2lf\n",Ans);
}
return 0;

}

poj1523 求关节点

发表于 2012-03-08   |   分类于 ACM   |  

##poj1523

题意就是说有个网络，找出这样的一个点，使网络变的不连通，并且要求给出去掉这个点后，网络被分成了多少块。
就是单纯的求关节点。
如果u是关节点则满足下面两个条件。
（1）如果u为深度优先搜索树的根，则u至少要有两个子孙。
（2）如果u不是根，则他至少有一个子女w，从w出发，不可能通过w及w的子孙通过一条回路到达u的祖先。（即当low[w] >= dfn[u]时，说明w及w的子孙不能达到u的祖先。）
